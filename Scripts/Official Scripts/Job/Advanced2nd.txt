// Advanced 2nd Class Quest/Changers - Non Player Character Locations
//  - Created by Tsusai
//
//	Authur's Note: These are by no means official,
//	but follow as much research as collected.
//	To go to Valhalla to turn into High Novice, find the
//	Scroll of Ymir somewhere in Nifel Heim, and find the NPC
//	that will take you there.  By default, since Nifel Heim
//	is a dangerous place, the character will die and will respawn
//	at the Novice training ground.  To become High 1st classes,
//	take the same quest as you were in a previously.  i.e. If you
//	were an assassin, you must take the theif quest.  At the time
//	you would be able to become 2-#, you need to return to Valhalla
//	via the same NPC.
//
//	Included as well in this script is a Prontera warper to Valhalla
//	that will grant you your appropriate High First Job, and warp you
//	again for the Adv 2nd change in Valhalla.  This skips the item finding quest.
//	That NPC is disabled by default.  Enable if you wish to use it.
//
// - NPCs at Valkyrie
//
//	"Kafra" - 53,32
//	Storage Capability
//
//	"Paladin" - 53,38
//	Changes Crusader into High Novice; Changes High Swordsman to Paladin
//
//	"Creator" - 53,42
//	Changes Alchemist into High Novice; Changes High Merchant to Creator
//
//	"Professor" - 53,46
//	Changes Sage into High Novice; Changes High Magician to Professor
//
//	"Clown" - 53,50
//	Changes Bard into High Novice; Changes Archer High to Clown
//
//	"Gypsy" - 53,54
//	Changes Dancer into High Novice; Changes Archer High to Gypsy
//
//	"Champion" - 53,56
//	Changes Monk into High Novice; Changes Acolyte High to Champion
//
//	"Stalker" - 53,58
//	Changes Rogue into High Novice; Changes Thief High to Stalker
//
//	"Transporter" - 44,32
//	Warps characters back to Nifelheim or Prontera for those using the Easy Changer
//
//	"Lord Knight" - 44,38
//	Changes Knight into High Novice; Changes High Swordsman to Lord Knight
//
//	"High Priest" - 44,42
//	Changes Priest into High Novice; Changes Acolyte High to High Priest
//
//	"High Wizard" - 44,46
//	Changes Wizard into High Novice; Changes High Magician to High Wizard
//
//	"Whitesmith" - 44,50
//	Changes Blacksmith into High Novice; Changes Merchant High to Whitesmith
//
//	"Sniper" - 44,55
//	Changes Hunter into High Novice; Changes Archer High to Sniper
//
//	"Assassin Cross" - 44,58
//	Changes Assassin into High Novice; Changes Thief High to Assassin Cross
//
// - Other NPCs
//
//	"Dusty Scroll" - 
//	Gives the Scroll of Ymir needed for teleport to Valkyrie
//
//	"?" - 
//	Takes the Scroll of Ymir and teleports character to Valkyrie
//
//	"Advanced Class Info" - Prontera: X,Y
//	Warps character to Valhalla for High Novice and Adv 2nd class change
//	Grants High First Class Change
//
//========================================================================================

//Valkyrie NPCs
//Right side going from south to north
valkyrie.gat,53,32,3	script	"Kafra"	117,{
	set @Kafra_COUPON = 0;
	cutin "Kafra_01",2;
	mes "[Kafra Personnel]";
	mes "Welcome to Kafra Corp.";
	mes "We will stay with you";
	mes "wherever you go.";
	next;
 L_START_1:
	check @Kafra_COUPON = 1,-,L_SUB_0_1;
		mes "[Kafra Personnel]";
		mes "How can I help you?";
		next;
 L_SUB_0_1:
	menu "Storage",L_MENU_2,"Cancel",L_MENU_7;
	L_MENU_2:
		getskilllevel 1,@storagecheck;
		check @storagecheck >= 6,L_SUB_2_1,-;
			mes "[Kafra Personnel]";
			mes "I am sorry but you have to be at least";
			mes "Novice level 6 if you want to use the storage.";
			check @Kafra_COUPON = 0,-,L_ROOT_2_1;
				goto L_EXIT;
		L_ROOT_2_1:
				next;
				goto L_START_1;
	L_SUB_2_1:
		check @Kafra_COUPON = 0,-,L_SUB_2_2;
			check Zeny < 40,-,L_ROOT_2_2;
				mes "[Kafra Personnel]";
				mes "Dear. you don't have enough Money.";
				mes "The Storage fee is 40 Zeny.";
				goto L_EXIT;
		L_ROOT_2_2:
			set Zeny -= 40,0;
			store;
			goto L_EXIT;
	L_SUB_2_2:
		store;
		goto L_EXIT;
	L_MENU_7:
		mes "[Kafra Personnel]";
		mes "We strive for perfection in our service.";
		mes "Rest assured that Kafra Corporation will always be there to help you.";
		mes "Thank you for your business.";
 L_EXIT:
		cutin "Kafra_01",255;
		close;
}

valkyrie.gat,53,38,3	script	"Paladin"	752,{
//Crusader->HighSword->Paladin
	check job = 14,Rebirth,-;
	check job = 25,Transform,-;
		mes "[Paladin]";
		mes "I'm sorry, but I cannot assist you.";
		mes "Ask another.";
		close;
	Rebirth:
		mes "[Paladin]";
		mes "Greetings my fellow Crusader";
		next;
		mes "Do you wish to be reborn so that you may become a Paladin?";
		next;
		menu "Yes Sir",-,"No Sir",Cancel;
			mes "[Paladin]";
			mes "Most Excelent";
			mes "I will now check to see if you are in deed worthy.";
			next;
			check baselevel >= 99,-,LowBlvl;
			check joblevel >= 50,-,LowJlvl1;
			check skillpoint = 0,-,SkPoint;
				mes "[Paladin]";
				mes "You meet my requirements.";
				mes "I will ask once more, are you sure you wish to relive your life, and return to me for the final phase?";
				next;
				menu "Yes Sir",-,"No Sir",Cancel;
					mes "[Paladin]";
					mes "Alright.";
					next;
					mes "[Paladin]";
					mes "Ashes to ashes";
					mes "Dust to dust";
					next;
					save "new_1-1",50,100;
					set REBIRTH = 14;
					option 0,0;
					option 1,0;
					option 2,0;
					speed 140;
					jobchange 24;
					basereset;
					resetstat;
					resetskill;
					set statuspoint = 100;
					set skillpoint = 0;
					setskilllevel 0x1,0;
					die;
					close;
			LowBlvl:
				mes "[Paladin]";
				mes "I am sorry, but you must have a base level of 99";
				ReturnHome:
					next;
					mes "[Paladin]";
					mes "Return home and train yourself";
					close;
			LowJlvl1:
				mes "[Paladin]";
				mes "I am sorry, but you must have a job level of 50";
				goto ReturnHome;
			SkPoint:
				mes "[Paladin]";
				mes "Please use up all your skill points";
				close;
	Transform:
			mes "[Paladin]";
			mes "Ah, you have returned.";
			mes "Are you ready to become a Paladin?";
			next;
			menu "Yes Sir",-,"No Sir",Cancel;
				mes "[Paladin];
				mes "I must check one last time to see if you can become a Paladin.";
				next;
				check joblevel >= 40,-,LowJlvl2;
				check skillpoint = 0,-,SkPoint;
					mes "[Paladin]";
					mes "Congradualations";
					mes "You pass my tests";
					next;
					option 0,0;
					option 1,0;
					option 2,0;
					jobchange 38;
					mes "[Paladin]";
					mes "Go and use your new skills.";
					close;
				LowJlvl2:
					mes "[Paladin]";
					mes "I am sorry, but you must have a job level of 40";
					goto ReturnHome;
	Cancel:
		mes "[Paladin]";
		mes "Go and follow your soul";
		mes "It'll lead you to true happiness";
		close;
}
